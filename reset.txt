Procedure PING: // Na svakih 1s zvati ovo
  int failures = 0; // Broj uzastopnih neuspeha
  try {
    response = this->next("/api/basic/ok"); // pozovi /api/basic/ok na NEXT cvoru
    if (response.status == "ok")  {
      failures = 0;
    }
  } catch {
    failures += 1;
    if (failures == 5) { // 5s proÅ¡lo
      try {
        this->prev("/api/basic/check", this->next) // proveri da li je tvoj NEXT ziv preko tvog PREV cvora
        failures = 0;
      } catch {
        // next ili prev cvor nije ziv :(
        call RECONFIGURE;
      }
    }
  }
END

Procedure RECONFIGURE:
  response = bootstrap/reset

  if (response.can_reset == false) RETURN; // rekonfiguracija je u toku

  BFS Broadcast /api/network/reset na svaki cvor u mrezi (i sebe!)

  this.reset  // resetuj sebe
END
